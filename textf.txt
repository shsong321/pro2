#include <stdio.h>
#include <time.h>

int random(int n1, int n2) { return rand() % n1 + n2; }

int main(void) {
  srand(time(NULL)); // 난수 초기화
  int correct_num = 0;
  int n1, n2, h, ans;
  while (correct_num < 10) {
    n1 = random(20, 1);
    n2 = random(20, 1);
    h = random(4, 0); // 0 1 2 3

    if (h == 0) {
      printf("[%d/10] %d + %d = ", correct_num, n1, n2);
      scanf("%d", &ans);
      if (ans == n1 + n2) {
        printf("correct!");
        correct_num++;
      } else {
        printf("Wrong! answer is %d", n1 + n2);
      }
    }
    if (h == 1) {
      printf("[%d/10] %d - %d = ", correct_num, n1, n2);
      scanf("%d", &ans);
      if (ans == n1 - n2) {
        printf("correct!");
        correct_num++;
      } else {
        printf("Wrong! answer is %d", n1 - n2);
      }
    }
    if (h == 2) {
      printf("[%d/10] %d * %d = ", correct_num, n1, n2);
      scanf("%d", &ans);
      if (ans == n1 * n2) {
        printf("correct!");
        correct_num++;
      } else {
        printf("Wrong! answer is %d", n1 * n2);
      }
    }
    if (h == 3) {
      printf("[%d/10] %d / %d = ", correct_num, n1, n2);
      scanf("%d", &ans);
      if (ans == n1 / n2) {
        printf("correct!");
        correct_num++;
      } else {
        printf("Wrong! answer is %d", n1 / n2);
      }
    }
    printf("\n");
  }
  printf("Clear!");

  return 0;
}
